#默认配置
var defaults = {

    autoPlay : true,                                 //是否自动播放
    	 	 
    Speed : 3000,                                    //自动执行间隔时间
         
    timed : 300,                                     //动画执行时间
         
    moveType : 'scroll',                             //切换方式默认是滚动 {scroll | opacity}
         
    mouseEvent : 'click',                            //按钮触发事件类型{mouseover | click} 
         
    btnShow : true,                                  //是否隐藏按钮        { true | false}
         
    btnBox : 'radio',                                //按钮class类名                       
         
    aBtn   : true                                    //是否隐藏两侧左右按钮{ true | false}   
    	 	 
};
